    <script>
        // 拖拽逻辑
        const dragItems = document.querySelectorAll('.drag-item');
        const dropTargets = document.querySelectorAll('.drop-target');
        const tipText = document.getElementById('tip');
        let correctCount = 0;

        // 拖拽开始
        dragItems.forEach(item => {
            item.addEventListener('dragstart', (e) => {
                e.dataTransfer.setData('text/plain', item.dataset.value);
                item.style.opacity = '0.6';
            });

            item.addEventListener('dragend', () => {
                item.style.opacity = '1';
            });
        });

        // 目标框事件
        dropTargets.forEach(target => {
            // 允许拖拽进入
            target.addEventListener('dragover', (e) => {
                e.preventDefault();
                target.classList.add('highlight');
            });

            // 离开目标框
            target.addEventListener('dragleave', () => {
                target.classList.remove('highlight');
            });

            // 放下模块
            target.addEventListener('drop', (e) => {
                e.preventDefault();
                target.classList.remove('highlight');
                
                const draggedValue = e.dataTransfer.getData('text/plain');
                const correctValue = target.dataset.correct;

                // 检查是否正确
                if (draggedValue === correctValue) {
                    // 正确：添加正确样式+动画
                    target.classList.add('correct', 'scale-in');
                    target.innerHTML = `
                        <div class="target-text">
                            <i class="fa fa-check-circle" style="color:#4ade80;"></i> 
                            ${target.querySelector('.target-text').textContent}
                        </div>
                    `;
                    // 禁用重复拖拽
                    target.style.pointerEvents = 'none';
                    correctCount++;
                    tipText.textContent = `✅ 正确！已完成 ${correctCount}/4 个步骤`;
                    tipText.style.color = '#4ade80';

                    // 全部正确提示
                    if (correctCount === 4) {
                        setTimeout(() => {
                            tipText.textContent = '🎉 恭喜！你已掌握AI批改的完整流程！';
                            tipText.style.fontSize = '18px';
                            tipText.style.textShadow = '0 0 10px rgba(74, 222, 128, 0.6)';
                        }, 800);
                    }
                } else {
                    // 错误：抖动动画+提示
                    const wrongItem = document.querySelector(`.drag-item[data-value="${draggedValue}"]`);
                    wrongItem.classList.add('shake');
                    setTimeout(() => wrongItem.classList.remove('shake'), 500);
                    
                    tipText.textContent = '❌ 不对哦，再想想这个步骤的核心是做什么？';
                    tipText.style.color = '#f87171';
                }
            });
        });
    </script>
